3.5 实例

group_cyl<-group_by(mpg,cyl)
data_sumr<- summarise(group_cyl,avg=mean(hwy),se=sd(hwy)/sqrt(n()))
data_sumr
# A tibble: 4 × 3
    cyl   avg    se
  <int> <dbl> <dbl>
1     4  28.8 0.502
2     5  28.8 0.25 
3     6  22.8 0.415
4     8  17.6 0.390
p<-ggplot(data_sumr,aes(cyl, avg)) + geom_line()+geom_point(size=4) 
p+geom_errorbar(aes(ymin=avg-se, ymax=avg+se), width=0.1)
